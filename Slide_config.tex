\newcommand{\insertpgf}[5][0.5]{
    \begin{figure}[H]
        #5
        \centering
%        {\pgfplotsset{#6}}
        \resizebox{#1\textwidth}{!}{\input{#2}}
        \caption*{#3}\label{#4}
    \end{figure}
}%用法:\insertpgf{発表/実験1.pgf}{実験1：各波長の$I$-$V$特性}{fig:exp1}{\vspace{-3em}}
\newcommand{\insertpng}[5][0.5]{
    \begin{figure}[H]
        #5
        \centering
        \includegraphics[width=#1\textwidth]{#2}
        \caption*{#3}\label{#4}
    \end{figure}
}

    \end{figure}
}
